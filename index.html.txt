<script>

let orders = [
{ id: 1001, service: "Facebook Like", link: "fb.com/post1", qty: 1000, done: 800, status: "processing" },
{ id: 1002, service: "TikTok Follow", link: "tiktok.com/user", qty: 500, done: 500, status: "completed" },
{ id: 1003, service: "YouTube View", link: "youtube.com/video", qty: 10000, done: 2000, status: "pending" }
];

function render() {
    let html = "";
    let search = document.getElementById("search").value.toLowerCase();
    let status = document.getElementById("status").value;

    orders.forEach(o => {
        if (
            (search && !o.link.includes(search) && !o.service.toLowerCase().includes(search)) ||
            (status && o.status !== status)
        ) return;

        html += `
        <div class="card">
            <div class="row"><b>#${o.id}</b> <span class="status ${o.status}">${o.status}</span></div>
            <div>Dịch vụ: ${o.service}</div>
            <div>Link: ${o.link}</div>
            <div>Tiến độ: ${o.done}/${o.qty}</div>
        </div>`;
    });

    document.getElementById("orders").innerHTML = html;
}

render();

document.getElementById("search").oninput = render;
document.getElementById("status").onchange = render;

</script>
